548/* Note that memset() is a byte assignment and this is a 32-bit assignment, so they're not directly equivalent. */
549SDL_FORCE_INLINE void SDL_memset4(void *dst, Uint32 val, size_t dwords)
550{
551#if defined(__GNUC__) && defined(__i386__)
552 int u0, u1, u2;
553 __asm__ __volatile__ (
554 "cld \n\t"
555 "rep ; stosl \n\t"
556 : "=&D" (u0), "=&a" (u1), "=&c" (u2)
557 : "0" (dst), "1" (val), "2" (SDL_static_cast(Uint32, dwords))
558 : "memory"
559 );
560#else
561 size_t _n = (dwords + 3) / 4;
562 Uint32 *_p = SDL_static_cast(Uint32 *, dst);
563 Uint32 _val = (val);
564 if (dwords == 0) {
565 return;
566 }
567 switch (dwords % 4) {
568 case 0: do { *_p++ = _val; SDL_FALLTHROUGH;
569 case 3: *_p++ = _val; SDL_FALLTHROUGH;
570 case 2: *_p++ = _val; SDL_FALLTHROUGH;
571 case 1: *_p++ = _val;
572 } while ( --_n );
573 }
574#endif
575}
